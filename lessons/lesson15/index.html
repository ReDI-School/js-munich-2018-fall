Function call: order matters

Function arguments are treated like variables inside the function


    function calculateDistance(speed, time) {
        return speed * time;
    }

    let carSpeed = 150; // km/h
    let travelDuration = 3; // hours
    let travelDistance = calculateDistance(carSpeed, travelDuration);

    alert("Distance travelled: " + travelDistance + " km");
